Quick Start

// Enhanced audio by default
const tts = new SimpleTTS();

// Speak with studio-quality audio
tts.onReady(() => {
  tts.speak('Hello world!', audioData => {
    SimpleTTS.playAudioData(audioData);
  });
});

Advanced Usage

// Control enhancement
tts.speak('Hello world!', {
  voice: 'en-us',     // Voice ID
  rate: 200,          // Speed (80-450)
  pitch: 75,          // Pitch (0-100)
  volume: 1.2,        // Volume (0-2.0)
  enhance: true       // Audio enhancement (default: true)
}, audioData => {
  // Play with optional reverb
  SimpleTTS.playAudioData(audioData, {
    reverb: true      // Add warmth
  });
});

Disable Enhancement (Raw Audio)

// For raw eSpeak-ng output
const tts = new SimpleTTS({
  enhanceAudio: false
});

// Or per-request
tts.speak('Hello', { enhance: false }, callback);
